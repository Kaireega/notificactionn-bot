def _convert_to_html(self, message: str) -> str:
    """Convert Markdown-style formatting to HTML for Telegram."""
    html_message = message
    html_message = html_message.replace('**', '<b>').replace('**', '</b>')
    html_message = html_message.replace('_', '<i>').replace('_', '</i>')
    return html_message
